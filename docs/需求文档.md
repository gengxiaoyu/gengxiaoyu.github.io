## 1. 项目概述

### 1.1 项目定位
**CodeVault** - 静态全栈开发者资源站

- **核心价值**：资源分享 + 个人品牌展示
- **技术架构**：纯静态，单页应用体验
- **开发原则**：模块化、高复用、低耦合
- **数据管理**：JSON + Markdown

### 1.2 技术栈

| 层级 | 技术选型 | 说明 |
|------|---------|------|
| **HTML** | HTML5 | 语义化标签 |
| **CSS** | CSS3 + CSS Variables | 统一设计系统 |
| **JavaScript** | ES6+ Modules | 模块化开发 |
| **构建工具** | Rollup（可选） | 代码打包 |
| **Markdown** | marked.js | 博客渲染 |
| **代码高亮** | Prism.js | 语法高亮 |
| **搜索** | fuse.js | 全文搜索 |

### 1.3 核心模块（MVP）

```
CodeVault
├── 首页（Home）- 简化版Hero
├── 资源库（Library）- 核心功能
├── 关于（About）- 个人介绍
└── 后续扩展（Blog、Portfolio、Docs）
```

#### 2.3.2 验证脚本（validation/validate.js）

```javascript
/**
 * 数据验证工具
 * 验证JSON数据格式和完整性
 */

class DataValidator {
  /**
   * 验证单个资源
   * @param {Object} resource 资源对象
   * @returns {Object} { valid: boolean, errors: string[] }
   */
  validateResource(resource) {
    const errors = [];

    // ID格式验证
    if (!/res-[0-9]{3}$/.test(resource.id)) {
      errors.push(`ID格式错误: ${resource.id}，应为 res-XXX 格式`);
    }

    // 必填字段验证
    const requiredFields = ['id', 'name', 'description', 'category'];
    requiredFields.forEach(field => {
      if (!resource[field]) {
        errors.push(`缺少必填字段: ${field}`);
      }
    });

    // URL格式验证
    if (resource.links) {
      resource.links.forEach((link, index) => {
        try {
          new URL(link.url);
        } catch (e) {
          errors.push(`链接 ${index + 1} URL格式错误: ${link.url}`);
        }
      });
    }

    return {
      valid: errors.length === 0,
      errors
    };
  }

  /**
   * 验证资源列表
   * @param {Array} resources 资源数组
   * @returns {Object} { valid: boolean, errors: Object }
   */
  validateResources(resources) {
    const errors = {};
    let valid = true;

    resources.forEach((resource, index) => {
      const result = this.validateResource(resource);
      if (!result.valid) {
        valid = false;
        errors[`res-${index + 1}`] = result.errors;
      }
    });

    return { valid, errors };
  }
}

export default DataValidator;
```

### 2.4 数据管理工具类

```javascript
// js/utils/data-manager.js
/**
 * 数据管理器
 * 负责加载、缓存、验证、更新JSON数据
 */

class DataManager {
  constructor() {
    this.cache = new Map();
    this.validator = new DataValidator();
    this.basePath = './data/';
  }

  /**
   * 加载数据文件
   * @param {string} filename 文件名
   * @param {boolean} useCache 是否使用缓存
   * @returns {Promise<Object>} 数据对象
   */
  async loadData(filename, useCache = true) {
    const cacheKey = filename;
    
    // 检查缓存
    if (useCache && this.cache.has(cacheKey)) {
      console.log(`从缓存加载: ${filename}`);
      return this.cache.get(cacheKey);
    }

    try {
      const response = await fetch(`${this.basePath}${filename}`);
      if (!response.ok) {
        throw new Error(`HTTP ${response.status}: ${filename}`);
      }

      const data = await response.json();
      
      // 验证数据
      const validation = this.validateData(filename, data);
      if (!validation.valid) {
        console.warn(`数据验证警告 (${filename}):`, validation.errors);
      }

      // 缓存数据
      this.cache.set(cacheKey, data);
      
      console.log(`成功加载: ${filename}`);
      return data;
    } catch (error) {
      console.error(`加载失败 (${filename}):`, error);
      throw error;
    }
  }

  /**
   * 批量加载数据
   * @param {Array<string>} filenames 文件名数组
   * @returns {Promise<Object>} 数据对象映射
   */
  async loadBatchData(filenames) {
    const promises = filenames.map(filename => 
      this.loadData(filename)
    );

    const results = await Promise.allSettled(promises);
    const data = {};

    results.forEach((result, index) => {
      const filename = filenames[index];
      if (result.status === 'fulfilled') {
        data[filename] = result.value;
      } else {
        console.error(`加载失败: ${filename}`, result.reason);
        data[filename] = null;
      }
    });

    return data;
  }
}

export default DataManager;
```
